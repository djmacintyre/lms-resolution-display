# String file for ResolutionDisplay plugin

PLUGIN_RESOLUTION_DISPLAY
	EN	Resolution Display

PLUGIN_RESOLUTION_DISPLAY_DESC
	EN	Adds title format tokens to display audio source resolution on player displays and screensavers.

PLUGIN_RESOLUTION_DISPLAY_SETTINGS
	EN	Resolution Display Settings

PLUGIN_RESOLUTION_DISPLAY_SHOW_LOSSLESS
	EN	Show lossless/lossy indicator

PLUGIN_RESOLUTION_DISPLAY_SHOW_LOSSLESS_DESC
	EN	Append a lossless indicator (e.g. "24/96 FLAC [L]") to the resolution string.

PLUGIN_RESOLUTION_DISPLAY_SHORT_FORMAT
	EN	Use short codec names

PLUGIN_RESOLUTION_DISPLAY_SHORT_FORMAT_DESC
	EN	Use abbreviated codec names (e.g. "FLC" instead of "FLAC", "MP3" instead of "mp3").

PLUGIN_RESOLUTION_DISPLAY_DSD_LABEL
	EN	DSD rate display style

PLUGIN_RESOLUTION_DISPLAY_DSD_LABEL_DESC
	EN	How to display DSD sample rates: as multiplier (DSD64, DSD128) or raw rate.

PLUGIN_RESOLUTION_DISPLAY_DSD_MULTIPLIER
	EN	Multiplier (DSD64)

PLUGIN_RESOLUTION_DISPLAY_DSD_RAW
	EN	Raw rate (2.8MHz)

PLUGIN_RESOLUTION_DISPLAY_SIGNAL_WARN_PCT
	EN	Signal warn threshold (%)

PLUGIN_RESOLUTION_DISPLAY_SIGNAL_WARN_PCT_DESC
	EN	HEALTH token shows degraded (◐ or !) when wireless signal drops below this percentage. Default: 40.

PLUGIN_RESOLUTION_DISPLAY_SIGNAL_GOOD_PCT
	EN	Signal good threshold (%)

PLUGIN_RESOLUTION_DISPLAY_SIGNAL_GOOD_PCT_DESC
	EN	HEALTH token shows good (● or OK) when wireless signal is at or above this percentage. Default: 60.

PLUGIN_RESOLUTION_DISPLAY_BUFFER_WARN_PCT
	EN	Buffer warn threshold (%)

PLUGIN_RESOLUTION_DISPLAY_BUFFER_WARN_PCT_DESC
	EN	HEALTH token shows degraded (◐ or !) when decode buffer fullness drops below this percentage. Default: 50.

PLUGIN_RESOLUTION_DISPLAY_HEALTH_GLYPHS
	EN	Use Unicode glyphs for health indicators

PLUGIN_RESOLUTION_DISPLAY_HEALTH_GLYPHS_DESC
	EN	Use Unicode glyphs (●◐○ and ▁▂▃▄▅▆▇█) for HEALTH and SIGNAL_BARS tokens. Disable for ASCII fallback (OK/!/!! and .:iI#) if your display cannot render them.
